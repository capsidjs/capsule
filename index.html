Mirroring example

on.input = ({ query }) => {
  query(".dest").textContent = query(".src").value;
};

Pubsub example

const EVENT = "my-event";

{
  const { on } = component("pub-element");

  on.click = ({ pub }) => {
    pub(EVENT, { hello: "clicked!" });
  };
}

{
  const { on, sub } = component("sub-element");

  sub(EVENT);

  on[EVENT] = ({ e, el }) => {
    el.textContent += " " + e.detail.hello;
  };
}
