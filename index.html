Mirroring example

const { on, query } = component("mirroring");

on.input = ({ query }) => {
  query(".dest").textContent = query(".src").value;
};

 

Hello example

const { on } = component("hello");

on.__mount__ = ({ el }) => {
  console.log(`Hello, I'm ${el.textContext}!`); // Alerts "Hello, I'm John Doe!"
};

Pubsub example



const EVENT = "my-event";

{
  const { on } = component("pub-element");

  on.click = ({ pub }) => {
    pub(EVENT, { hello: "world!" });
  };
}

{
  const { on, sub } = component("sub-element");

  sub(EVENT);

  on[EVENT] = ({ e }) => {
    console.log(e.detail.hello); // => world!
  };
}
